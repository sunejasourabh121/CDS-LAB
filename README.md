```Sourabh Suneja
#include <iostream>
using namespace std;

int main() {
    int num1, num2, sum = 0, diff = 0;
    cout << "Enter first number: ";
    cin >> num1;
    cout << "Enter the second number: ";
    cin >> num2;

    sum = num1 + num2;
    cout << "Sum of given numbers = " << sum << endl;

    diff = num1 - num2;
    cout << "Difference of given numbers = " << diff << endl;

    return 0;
}
```
